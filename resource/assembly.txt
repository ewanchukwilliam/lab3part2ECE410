ASSEMBLY INSTRUCTIONS:

# Address 0: Load test data
lw x1, 100(x0)      # x1 = 16 (load from memory address 100)

# Address 4: Load test data
lw x2, 104(x0)      # x2 = -8 (load from memory address 104)

# Address 8: Arithmetic - ADD
add x3, x1, x2      # x3 = x1 + x2 = 16 + (-8) = 8

# Address 12: Arithmetic - SUB
sub x4, x1, x2      # x4 = x1 - x2 = 16 - (-8) = 24

# Address 16: Immediate - ADDI
addi x5, x1, 4      # x5 = x1 + 4 = 16 + 4 = 20

# Address 20: Shift Right Arithmetic Immediate (positive number)
srai x6, x1, 2      # x6 = x1 >> 2 = 16 >> 2 = 4

# Address 24: Shift Right Arithmetic Immediate (negative number)
srai x7, x2, 1      # x7 = x2 >> 1 = -8 >> 1 = -4 (sign extends)

# Address 28: Store word
sw x3, 108(x0)      # Store x3 (8) to memory address 108

# Address 32: Branch NOT taken
beq x1, x2, 8       # Compare x1 == x2? (16 == -8?) NO, continue to next

# Address 36: Execute if branch not taken
addi x8, x1, 1      # x8 = x1 + 1 = 16 + 1 = 17 (EXECUTES)

# Address 40: Branch TAKEN
beq x1, x1, 8       # Compare x1 == x1? (16 == 16?) YES, branch to address 48

# Address 44: Should be skipped
addi x9, x0, 99     # x9 = 0 + 99 = 99 (SKIPPED due to branch)

# Address 48: Branch target
addi x9, x2, 2      # x9 = x2 + 2 = -8 + 2 = -6 (EXECUTES after branch)

# Address 52: Halt
halt                # Stop execution
